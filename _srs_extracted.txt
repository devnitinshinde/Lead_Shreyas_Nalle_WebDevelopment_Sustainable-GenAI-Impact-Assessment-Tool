2.  Overall  Description  
2.1  Technology  Stack  
Frontend:  Next.js  14  (App  Router),  React,  Tailwind  CSS,  Recharts  (for  dashboard  charts)  
Backend:  Next.js  API  routes  (JavaScript)  —  handles  proxy  forwarding,  key  management,  
dashboard
 
data,
 
report
 
generation
 
Authentication:  Firebase  Authentication  (email/password)  
Database:  Firebase  Firestore  (NoSQL)  —  stores  org  data,  keys,  call  logs,  projects,  reports,  
recommendations
 
File  Storage:  Firebase  Storage  —  stores  generated  PDF  reports  
Scheduled  Jobs:  Firebase  Cloud  Functions  with  scheduled  triggers  —  monthly  report  
generation,
 
daily
 
recommendation
 
computation
 
Email:  Firebase  Extensions  (Trigger  Email)  or  a  direct  SendGrid  API  call  from  a  Cloud  Function  

PDF  Generation:  pdfkit  (Node.js  library)  or  jsPDF  —  called  from  a  Next.js  API  route  or  Cloud  
Function
 
Encryption:  Node.js  built-in  crypto  module  (AES-256-GCM)  for  provider  key  encryption  
Hosting:  Vercel  (Next.js)  +  Firebase  (backend  services)

3.  Firestore  Data  Model  
This  replaces  the  SQL  schema  section.  Firestore  is  document-based.  Here  is  the  complete  
collection
 
structure.
 
Collection:  orgs  

One  document  per  registered  organisation.  
orgs/{orgId}  
  name:             string        //  "Acme  Corp"  
  email:            string        //  login  email,  same  as  Firebase  Auth  email  
  firebaseUid:      string        //  Firebase  Auth  UID,  used  to  link  auth  to  org  
  region:           string        //  "india"  |  "us"  |  "eu"  |  "norway"  
  industry:         string        //  "fintech"  |  "healthtech"  |  "ecommerce"  |  ...  
  size:             string        //  "1-10"  |  "11-50"  |  "51-200"  |  "200+"  
  aiProviders:      array         //  ["openai",  "anthropic",  "google"]  
  plan:             string        //  "free"  |  "pro"  
  createdAt:        timestamp  
  updatedAt:        timestamp  
 
Collection:  ecoKeys  
One  document  per  EcoTrack  API  key  issued  to  the  org.  
ecoKeys/{keyId}  
  orgId:            string        //  reference  to  orgs/{orgId}  
  projectId:        string|null   //  null  means  global  key  
  keyHash:          string        //  bcrypt  hash  of  the  full  key  —  never  store  plaintext  
  keyPrefix:        string        //  first  12  chars,  shown  in  UI  e.g.  "eco-sk-org12"  
  name:             string        //  user-given  label  e.g.  "Production"  
  revoked:          boolean  
  createdAt:        timestamp  
  lastUsedAt:       timestamp|null  

 
Collection:  providerKeys  
One  document  per  AI  provider  key  the  org  connects.  
providerKeys/{keyId}  
  orgId:            string  
  provider:         string        //  "openai"  |  "anthropic"  |  "google"  
  encryptedKey:     string        //  AES-256-GCM  encrypted  key  
  iv:               string        //  AES  initialisation  vector  (hex)  
  authTag:          string        //  AES  GCM  auth  tag  (hex)  
  nickname:         string        //  "Production  OpenAI  key"  
  createdAt:        timestamp  
 
Collection:  projects  
projects/{projectId}  
  orgId:            string  
  name:             string  
  description:      string  
  environment:      string        //  "production"  |  "staging"  |  "development"  
  archived:         boolean  
  createdAt:        timestamp  
 
Collection:  callLogs  
One  document  per  API  call  proxied.  This  is  the  most  important  collection  —  everything  on  the  
dashboard
 
comes
 
from
 
here.
 

callLogs/{logId}  
  orgId:            string  
  projectId:        string|null  
  provider:         string        //  "openai"  |  "anthropic"  |  "google"  
  model:            string        //  "gpt-4"  |  "gpt-3.5-turbo"  |  ...  
  promptTokens:     number  
  completionTokens:  number  
  totalTokens:      number  
  energyWh:         number        //  calculated  
  co2Grams:         number        //  calculated  
  region:           string        //  copied  from  org  at  time  of  call  
  status:           string        //  "success"  |  "failed"  
  createdAt:        timestamp  
 
Collection:  reports  
reports/{reportId}  
  orgId:            string  
  projectId:        string|null  
  periodStart:      timestamp  
  periodEnd:        timestamp  
  format:           string        //  "pdf"  |  "csv"  
  storageUrl:       string        //  Firebase  Storage  download  URL  
  generatedAt:      timestamp  
 

Collection:  recommendations  
recommendations/{recId}  
  orgId:            string  
  type:             string        //  "model_switch"  |  "context_length"  |  "staging"  
  title:            string  
  description:      string  
  energySavingWh:   number  
  co2SavingGrams:   number  
  dismissed:        boolean  
  dismissedAt:      timestamp|null  
  createdAt:        timestamp  
 
Subcollection:  modelRegistry (top-level  collection)  
modelRegistry/{modelId}  
  modelName:        string        //  "gpt-4"  
  provider:         string        //  "openai"  
  whPer1kTokens:    number        //  5.0  
  taskTags:         array         //  ["complex  reasoning",  "code  generation"]  
  updatedAt:        timestamp  
 
Collection:  regionCarbonIntensity (top-level  collection)  
regionCarbonIntensity/{regionId}  
  regionCode:       string        //  "india"  
  regionName:       string        //  "India"  

  gco2PerWh:        number        //  0.82  
  updatedAt:        timestamp

FR-27  Energy  Calculation  Logic  
//  In  the  proxy  API  route,  after  receiving  OpenAI  response:  
 
const  usage  =  responseBody.usage  
const  totalTokens  =  usage.total_tokens  
const  model  =  requestBody.model  
 
//  Look  up  from  modelRegistry  collection  in  Firestore  
const  modelDoc  =  await  db.collection('modelRegistry')  
                          .where('modelName',  '==',  model)  
                          .get()  
const  whPer1k  =  modelDoc.docs[0].data().whPer1kTokens  
 
//  Look  up  carbon  intensity  for  org's  region  
const  regionDoc  =  await  db.collection('regionCarbonIntensity')  
                           .where('regionCode',  '==',  org.region)  
                           .get()  
const  gco2PerWh  =  regionDoc.docs[0].data().gco2PerWh  

 
const  energyWh  =  (totalTokens  /  1000)  *  whPer1k  
const  co2Grams  =  energyWh  *  gco2PerWh  
 
//  Write  to  callLogs

  orgId,  projectId,  model,  provider,

FR-28  Proxy  Encryption  Implementation  
//  utils/encryption.js  
 
const  crypto  =  require('crypto')  
const  ALGORITHM  =  'aes-256-gcm'  
const  KEY  =  Buffer.from(process.env.ENCRYPTION_KEY,  'hex')  //  32-byte  hex  key  
 

function  encrypt(plaintext)  {  
  const  iv  =  crypto.randomBytes(12)  
  const  cipher  =  crypto.createCipheriv(ALGORITHM,  KEY,  iv)  
  const  encrypted  =  Buffer.concat([  
    cipher.update(plaintext,  'utf8'),  
    cipher.final()  
  ])  
  const  authTag  =  cipher.getAuthTag()  
  return  {  
    encryptedKey:  encrypted.toString('hex'),  
    iv:  iv.toString('hex'),  
    authTag:  authTag.toString('hex')  
  }  
}  
 
function  decrypt(encryptedKey,  iv,  authTag)  {  
  const  decipher  =  crypto.createDecipheriv(  
    ALGORITHM,  KEY,  Buffer.from(iv,  'hex')  
  )  
  decipher.setAuthTag(Buffer.from(authTag,  'hex'))  
  const  decrypted  =  Buffer.concat([  
    decipher.update(Buffer.from(encryptedKey,  'hex')),  
    decipher.final()  
  ])  

  return  decrypted.toString('utf8')  
}  
 
module.exports  =  {  encrypt,  decrypt  }
